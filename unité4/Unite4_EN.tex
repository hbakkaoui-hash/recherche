\section*{Appendix --- Numerical checks: GR (Schwarzschild) and QM (displaced oscillator)}
\small
The two calculations below quantify the \emph{two limits} of the ``RG/QM consistency'' section: general relativity recovered at frozen field ($\sim10^{-43}$), quantum mechanics recovered at $\varepsilon\to1$ ($\sim10^{-22}$). Data: $G=6.674\times10^{-11}$, $c=2.998\times10^{8}$, $M_\odot=1.989\times10^{30}$ (SI); electron $m=9.11\times10^{-31}$~kg, $q=-e=-1.602\times10^{-19}$~C, $\hbar=1.055\times10^{-34}$~J$\cdot$s.

\paragraph{A.1 Schwarzschild gravitational redshift (GR limit).}
Black hole $M=3M_\odot=5.967\times10^{30}$~kg; two static observers at $R_A=12$~km and $R_B=100$~km; $\nu_A=10^{15}$~Hz.
\emph{(1) Schwarzschild radius.} $R_S=2GM/c^2=8.86$~km.
\emph{(2) Position.} $R_A,R_B>R_S$: both observers are outside the horizon ($R_S/2R_A\approx0.37$, $A$ very compact).
\emph{(3) Redshift.} The static proper time is $d\tau=\sqrt{1-R_S/r}\,dt$, and $dt$ is the same along the radial photon geodesic (stationary metric), hence
\begin{equation}
\frac{\nu_B}{\nu_A}=\frac{d\tau_A}{d\tau_B}=\sqrt{\frac{1-R_S/R_A}{1-R_S/R_B}}. \tag{A.1}
\end{equation}
\emph{(4) Numerics.} $R_S/R_A=0.738$, $R_S/R_B=0.0886$; $\nu_B/\nu_A=\sqrt{0.262/0.911}=\sqrt{0.287}\approx\boxed{0.536}$, i.e.\ $\nu_B\approx5.36\times10^{14}$~Hz (green $\lambda\!\approx\!500$~nm seen as orange $\lambda\!\approx\!560$~nm: redshift).
\emph{(5--6) Dilation.} $d\tau_B/d\tau_A=1/0.536\approx\boxed{1.87}$: time runs $1.87\times$ faster for $B$. One signal per proper second of $A$ is received every $\Delta\tau_B\approx1.87$~s; after 100 emitted, $B$ has received only $\approx53$.
\emph{(7) Limit $R_A\to R_S$.} $\sqrt{1-R_S/R_A}\to0\Rightarrow\nu_B/\nu_A\to0$: infinite redshift at the horizon; $A$'s proper time ``freezes'' as seen from outside, while $A$ crosses the horizon in finite proper time.
\emph{Link to the framework.} The frozen-field branch $\psi\equiv-\pi$ is exactly Schwarzschild--de Sitter: these three values ($R_S$, $\nu_B/\nu_A$, $d\tau_B/d\tau_A$) are returned identically, the residual gap ($\sim10^{-43}$) coming from the de Sitter term alone. \st{proof}\st{numerical}

\paragraph{A.2 Displaced harmonic oscillator in a uniform field (QM limit).}
$\hat H=\hat p^2/2m+\tfrac12 m\omega^2\hat x^2-q\mathcal{E}\hat x$, with $\omega=10^{15}$~rad/s, $\mathcal{E}=10^{6}$~V/m.
\emph{(1) Normal form.} With $\hat x=\sqrt{\hbar/2m\omega}\,(\hat a+\hat a^\dagger)$,
$\hat H=\hbar\omega(\hat a^\dagger\hat a+\tfrac12)-q\mathcal{E}\sqrt{\hbar/2m\omega}\,(\hat a+\hat a^\dagger)$. The displacement $\hat a'=\hat a-\alpha$, $\alpha=q\mathcal{E}/\sqrt{2m\hbar\omega^3}$ (Weyl translation operator --- the \emph{same} conformal hinge as $\Omega=\sqrt f$ on the gravity side), gives
\begin{equation}
\hat H=\hbar\omega\Big(\hat a'^\dagger\hat a'+\tfrac12\Big)-\frac{q^2\mathcal{E}^2}{2m\omega^2},\qquad
E_n=\hbar\omega\Big(n+\tfrac12\Big)-\frac{q^2\mathcal{E}^2}{2m\omega^2}. \tag{A.2}
\end{equation}
The spectrum is rigidly shifted, levels remaining equally spaced by $\hbar\omega$.
\emph{(2) Energy shift.} $\Delta E=-q^2\mathcal{E}^2/(2m\omega^2)\approx-1.41\times10^{-26}$~J $\approx-8.8\times10^{-8}$~eV; since $\hbar\omega\approx0.66$~eV, $|\Delta E|/\hbar\omega\approx1.3\times10^{-7}$ (perturbative regime).
\emph{(3) Displaced center.} $V_{\rm eff}'=m\omega^2x-q\mathcal{E}=0\Rightarrow x_0=q\mathcal{E}/(m\omega^2)=\boxed{-0.176~\text{pm}}$ (the electron, negative charge, shifts opposite to the field); the frequency stays $\omega$.
\emph{(4) Coherent ground state.} $\hat a'|0'\rangle=0\Rightarrow\hat a|0'\rangle=\alpha|0'\rangle$: $|0'\rangle=|\alpha\rangle=e^{-|\alpha|^2/2}\sum_n(\alpha^n/\sqrt{n!})|n\rangle$, with $\langle\hat a^\dagger\hat a\rangle=|\alpha|^2=q^2\mathcal{E}^2/(2m\hbar\omega^3)\approx\boxed{1.34\times10^{-7}}\ll1$.
\emph{(5) Mean position.} $\langle0'|\hat x|0'\rangle=\sqrt{\hbar/2m\omega}\,(2\alpha)=q\mathcal{E}/(m\omega^2)=x_0$.
\emph{(6) Susceptibility.} $P=q\langle\hat x\rangle=q^2\mathcal{E}/(m\omega^2)$, hence $\chi=\partial P/\partial\mathcal{E}=q^2/(m\omega^2)\approx\boxed{2.82\times10^{-38}~\text{F}\cdot\text{m}^2}$ (consistent with an atomic polarizability, $\sim2.5\times10^{-3}\,a_0^3$).
\emph{Link to the framework.} Through $\hat H_{\rm eff}=\varepsilon\hat H_{\rm sys}$, only the energies are multiplied by $\varepsilon$; the \emph{structural} quantities ($x_0$, $\langle n\rangle$, $\chi$) are unchanged. At $\varepsilon\to1$ standard quantum mechanics is recovered exactly (gap $\sim10^{-22}$, cf.\ the RG/QM consistency section). \st{proof}\st{numerical}
